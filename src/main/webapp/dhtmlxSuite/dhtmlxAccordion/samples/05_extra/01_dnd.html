<script>
var dhxAccord,
logObj;
function doOnLoad() {
    dhxAccord = new dhtmlXAccordion("accordObj");
    dhxAccord.enableMultiMode();
    dhxAccord.addItem("a1", "a1: Main Page");
    dhxAccord.addItem("a2", "a2: Site Navigation");
    dhxAccord.addItem("a3", "a3: Support & Feedback");
    dhxAccord.addItem("a4", "a4: Comments");
    // use this method only after you will add all needed items into accordion;
    dhxAccord.enableDND();
    // attach event;
    dhxAccord.attachEvent("onDrop", function(id, ind_old, ind_new) {
        if (!logObj)
            logObj = document.getElementById("logObj");
        logObj.innerHTML += "onDrop event, id: '" + id + "', from index " + ind_old + " to index " + ind_new + "<br>";
        logObj.scrollTop = logObj.offsetHeight;
    });
}
</script>
