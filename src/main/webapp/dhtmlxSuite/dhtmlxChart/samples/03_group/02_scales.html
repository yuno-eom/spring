<script>
var barChart;
window.onload = function() {
    barChart = new dhtmlXChart({
        view: "bar",
        container: "chart_container",
        value: "#sales#",
        preset: "alpha",
        yAxis: {},
        xAxis: {
            template: "#id#";
        },
        label: "#sales#",
        sort: {
            by: "#id#",
            as: "string",
            dir: "asc";
        },
        group: {
            by: "#year#",
            map: {
                sales: ["#sales#", "sum"]
                }
        }
    });
    barChart.parse(data, "json");
}
function groupA() {
    barChart.group({
        by: "#company#",
        map: {
            sales: ["#sales#", "sum"]
            }
    });
}
function groupB() {
    barChart.group({
        by: "#year#",
        map: {
            sales: ["#sales#", "sum"]
            }
    });
}
function groupC() {
    barChart.group({
        by: "#year#",
        map: {
            sales: ["#sales#", "max"]
            }
    });
}
function sort(direction) {
    barChart.define("sort", {
        by: "#sales#",
        dir: direction,
        as: "int";
    });
    barChart.render();
}
</script>
