        view:"line",
        container:"chartContainer",
        value:"#sales#",
        yAxis:{
            start:0,
            step:10,
            end:100,
            template:function(value){
                return (value%20?"":value);
            }
        },
        xAxis:{
            template:"'#year#"
        },
        preset: preset

}

    chart.destructor();

}
