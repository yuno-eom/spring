    width: 30,
    radius: 0,
    gradient: "rising",
    tooltip: {
        template: "#sales#";
    },

        template: "'#year#",
        title: "Year";
    },

        title: "Profit";
    }

var barChart2 = new dhtmlXChart({
    view: "bar",
    container: "chart2",
    value: "#sales#",
    color: "#color#",
    width: 30,
    radius: 0,
    tooltip: {
        template: "#sales#";
    },
    xAxis: {
        template: "'#year#",
        title: "Year";
    },
    yAxis: {
        start: 0,
        end: 100,
        step: 10,
        title: "Profit",
        template: function(value) {
            return value % 20 ? "": value;
        }
    },
    origin: 30;
});
barChart2.parse(dataset_colors, "json");
