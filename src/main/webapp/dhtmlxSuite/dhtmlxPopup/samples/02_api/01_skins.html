var myToolbar;
var myPop;
 
function doOnLoad(skin) {
 
    if (!skin)
        skin = "dhx_skyblue";
 
    myToolbar = new dhtmlXToolbarObject("myToolbar", skin);
    myToolbar.setIconsPath("../common/imgs/");
    myToolbar.loadXML("../common/toolbar.xml?etc=" + new Date().getTime(), function() {
 
        myToolbar.setItemText("workbut", skin);
 
        myPop = new dhtmlXPopup({
            toolbar: myToolbar,
            id: "workbut"
        });
 
        myPop.attachList("name", [{
            id: "dhx_skyblue",
            name: "dhx_skyblue",
            price: "&euro; 31,550"
        }, {
            id: "dhx_web",
            name: "dhx_web",
            price: "&euro; 30,990"
        }, {
            id: "dhx_terrace",
            name: "dhx_terrace",
            price: "&euro; 30,990"
        }]);
 
        myPop.attachEvent("onClick", function(id) {
            myPop.unload();
            myToolbar.unload();
            doOnLoad(id);
        });
    });
 
}
</script>
