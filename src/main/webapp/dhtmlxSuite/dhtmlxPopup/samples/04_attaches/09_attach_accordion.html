var myToolbar;
var myPop;
var myAcc;
 
function doOnLoad() {
 
    myToolbar = new dhtmlXToolbarObject("myToolbar", "dhx_skyblue");
    myToolbar.setIconsPath("../common/imgs/");
    myToolbar.loadXML("../common/toolbar.xml?etc=" + new Date().getTime(), function() {
 
        myToolbar.setItemText("workbut", "Accordion");
 
        myPop = new dhtmlXPopup({
            toolbar: myToolbar,
            id: "workbut"
        });
        myPop.attachEvent("onShow", function() {
            if (!myAcc) {
                myAcc = myPop.attachAccordion(350, 300);
                myAcc.addItem("a1", "General Info");
                myAcc.addItem("a2", "Common Setting");
                myAcc.addItem("a3", "Help &amp; Support");
                myAcc.openItem("a1");
            }
        });
 
    });
 
}
</script>
