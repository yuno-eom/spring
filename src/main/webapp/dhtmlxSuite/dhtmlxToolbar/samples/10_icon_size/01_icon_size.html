<script>
var dhxToolbar,
dhxToolbar2;
function doOnLoad() {
    dhxToolbar = new dhtmlXToolbarObject("toolbarObj", "dhx_skyblue");
    loadToolbar(18);
    //;
    dhxToolbar2 = new dhtmlXToolbarObject("toolbarObj2", "dhx_web");
    loadToolbar2(18);
}
function loadToolbar(a) {
    dhxToolbar.clearAll();
    dhxToolbar.setIconSize(a);
    dhxToolbar.setIconsPath("../common/icon_size/" + a + "/");
    dhxToolbar.loadXML("../common/dhxtoolbar_button2.xml?etc=" + new Date().getTime());
}
function loadToolbar2(a) {
    dhxToolbar2.clearAll();
    dhxToolbar2.setIconSize(a);
    dhxToolbar2.setIconsPath("../common/icon_size/" + a + "/");
    dhxToolbar2.loadXML("../common/dhxtoolbar_button2.xml?etc=" + new Date().getTime());
}
</script>
